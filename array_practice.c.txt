#include <stdio.h>

void display(int arr[], int n) {
    for(int i = 0; i < n; i++) printf("%d ", arr[i]);
    printf("\n");
}

int main() {
    int arr[10] = {10, 20, 30, 40, 50}; 
    int size = 5;

    printf("--- Array Operations ---\n");
    printf("Original: ");
    display(arr, size);

    // INSERTION
    int pos = 2, element = 25;
    for (int i = size; i >= pos; i--) { arr[i] = arr[i - 1]; }
    arr[pos] = element;
    size++;
    printf("After Insertion: ");
    display(arr, size);

    // DELETION
    int del_pos = 3;
    for (int i = del_pos; i < size - 1; i++) { arr[i] = arr[i + 1]; }
    size--;
    printf("After Deletion: ");
    display(arr, size);

    return 0;
}
